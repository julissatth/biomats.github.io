import java.awt.image.BufferedImage; import java.io.File; import java.io.IOException; import javax.imageio.ImageIO; public class ImageToCode { public static void main(String[] args) { try { BufferedImage image = ImageIO.read(new File("ruta/de/la/imagen.jpg")); int width = image.getWidth(); int height = image.getHeight(); System.out.println("BufferedImage image = new BufferedImage(" + width + ", " + height + ", BufferedImage.TYPE_INT_RGB);"); System.out.println("Graphics2D g2d = image.createGraphics();"); for (int y = 0; y < height; y++) { for (int x = 0; x < width; x++) { int rgb = image.getRGB(x, y); System.out.println("g2d.setColor(new Color(" + (rgb >> 16 & 0xFF) + ", " + (rgb >> 8 & 0xFF) + ", " + (rgb & 0xFF) + "));"); System.out.println("g2d.drawLine(" + x + ", " + y + ", " + x + ", " + y + ");"); } } System.out.println("g2d.dispose();"); System.out.println("ImageIO.write(image, \"jpg\", new File(\"ruta/de/la/imagen/convertida.jpg\"));"); } catch (IOException e) { e.printStackTrace(); } } }
